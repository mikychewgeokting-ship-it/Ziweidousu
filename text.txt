<script>
lucide.createIcons();

const hoursData=[
{name:"子時",range:"23:00-00:59",value:"子時"},{name:"丑時",range:"01:00-02:59",value:"丑時"},
{name:"寅時",range:"03:00-04:59",value:"寅時"},{name:"卯時",range:"05:00-06:59",value:"卯時"},
{name:"辰時",range:"07:00-08:59",value:"辰時"},{name:"巳時",range:"09:00-10:59",value:"巳時"},
{name:"午時",range:"11:00-12:59",value:"午時"},{name:"未時",range:"13:00-14:59",value:"未時"},
{name:"申時",range:"15:00-16:59",value:"申時"},{name:"酉時",range:"17:00-18:59",value:"酉時"},
{name:"戌時",range:"19:00-20:59",value:"戌時"},{name:"亥時",range:"21:00-22:59",value:"亥時"}
];

const hourSelector=document.getElementById('hour-selector');
const selectedHourInput=document.getElementById('selected-hour');

hoursData.forEach(hour=>{
const card=document.createElement('div');
card.className='hour-card';
card.innerHTML=`<div class="text-yellow-400 font-bold text-sm mb-1">${hour.name}</div><div class="text-xs text-gray-400 font-mono">${hour.range}</div>`;
card.onclick=()=>{
document.querySelectorAll('.hour-card').forEach(c=>c.classList.remove('active'));
card.classList.add('active');
selectedHourInput.value=`${hour.value} (${hour.range})`;
};
hourSelector.appendChild(card);
});

function showToast(msg){
const toast=document.getElementById('toast');
toast.textContent=msg;
toast.classList.add('show');
setTimeout(()=>toast.classList.remove('show'),3000);
}

function copyAllText(){
const texts=["中華第一神術·紫微斗數","專屬命主命盤","根據出生時辰精準排盤","立即解析完整內容","完整版詳細命理解讀"];
navigator.clipboard.writeText(texts.join('\n')).then(()=>showToast('文案已複製！')).catch(()=>showToast('複製失敗'));
}

function startDivination(){
const name=document.getElementById('name').value;
const gender=document.querySelector('input[name="gender"]:checked')?.value;
const year=document.getElementById('year').value;
const month=document.getElementById('month').value;
const day=document.getElementById('day').value;
const hour=selectedHourInput.value;

if(!name||!year||!month||!day||!hour){showToast('請填寫完整的出生資訊！');return;}

document.getElementById('loading-tip').classList.remove('hidden');
setTimeout(()=>{
document.getElementById('res-name').textContent=name;
document.getElementById('res-gender').textContent=gender;
document.getElementById('res-birth').textContent=`${year}年${month}月${day}日 ${hour}`;
document.getElementById('loading-tip').classList.add('hidden');
document.getElementById('result-modal').classList.remove('hidden');
lucide.createIcons();
},2000);
}

function closeModal(){document.getElementById('result-modal').classList.add('hidden');}
